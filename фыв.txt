import React, { useState, useEffect } from 'react';
import {
  TrendingUp, Package, DollarSign, Calendar, Search, ShoppingCart, Printer, BarChart3,
  FileText, Calculator, Clock, Check, X, Edit2, Trash2, Plus, ArrowLeft, ArrowRight,
  Download
} from 'lucide-react';

const ElectroMaxPro = () => {
  const [products, setProducts] = useState([]);
  const [searchTerm, setSearchTerm] = useState('');
  const [cart, setCart] = useState([]);
  const [orders, setOrders] = useState([]);
  const [editingProduct, setEditingProduct] = useState(null); // table edit
  const [cardEditingId, setCardEditingId] = useState(null); // inline card edit
  const [cardEditForm, setCardEditForm] = useState({ name: '', price: '', unit: '' });
  const [editForm, setEditForm] = useState({ name: '', price: '', unit: '' }); // table edit form
  const [currentView, setCurrentView] = useState('order');
  const [loading, setLoading] = useState(true);
  const [currentPage, setCurrentPage] = useState(1);
  const [exchangeRate, setExchangeRate] = useState(12650);
  const [selectedCurrency, setSelectedCurrency] = useState('USD');
  const [calculatorAmount, setCalculatorAmount] = useState('');
  const [dateFilter, setDateFilter] = useState({ start: '', end: '' });
  const [customerInfo, setCustomerInfo] = useState({ name: '', phone: '' });
  const [showCalculator, setShowCalculator] = useState(false);
  const [storageReady, setStorageReady] = useState(false);
  const [showAllProducts, setShowAllProducts] = useState(true);
  const itemsPerPage = 50;

  const ORDERS_STORAGE_KEY = 'electromax_orders';

  useEffect(() => {
    const saved = localStorage.getItem(ORDERS_STORAGE_KEY);
    if (saved) {
      try { setOrders(JSON.parse(saved)); } catch { setOrders([]); }
    }
    setStorageReady(true);
  }, []);

  useEffect(() => {
    if (storageReady) localStorage.setItem(ORDERS_STORAGE_KEY, JSON.stringify(orders));
  }, [orders, storageReady]);

  useEffect(() => {
    const loadProducts = async () => {
      try {
        setLoading(true);
        const response = await fetch('/products.json');
        if (!response.ok) throw new Error('File not found');
        const data = await response.json();
        setProducts(data.products || data);
      } catch (error) {
        console.error('Error loading products.json:', error);
        setProducts(generateDemoProducts());
      } finally {
        setLoading(false);
      }
    };

    loadProducts();
    fetchExchangeRate();
    const interval = setInterval(fetchExchangeRate, 300000);
    return () => clearInterval(interval);
  }, []);

  const generateDemoProducts = () => {
    const categories = ['–≠–ª–µ–∫—Ç—Ä–æ–Ω–∏–∫–∞', '–ë—ã—Ç–æ–≤–∞—è —Ç–µ—Ö–Ω–∏–∫–∞', '–ö–æ–º–ø—å—é—Ç–µ—Ä—ã', '–ê–∫—Å–µ—Å—Å—É–∞—Ä—ã', '–¢–µ–ª–µ—Ñ–æ–Ω—ã'];
    const ps = [];
    for (let i = 1; i <= 100; i++) {
      ps.push({
        id: i,
        name: `${categories[i % 5]} –¢–æ–≤–∞—Ä ${i}`,
        price: parseFloat((Math.random() * 1000 + 50).toFixed(2)),
        unit: '—à—Ç',
        category: categories[i % 5],
        stock: Math.floor(Math.random() * 100)
      });
    }
    return ps;
  };

  const clearOldOrders = (days = 30) => {
    const cutoff = new Date();
    cutoff.setDate(cutoff.getDate() - days);
    const filtered = orders.filter(order => new Date(order.date) >= cutoff);
    const removedCount = orders.length - filtered.length;
    if (removedCount === 0) {
      alert('–°—Ç–∞—Ä—ã—Ö –∑–∞–∫–∞–∑–æ–≤ –¥–ª—è —É–¥–∞–ª–µ–Ω–∏—è –Ω–µ—Ç');
      return;
    }
    if (window.confirm(`–£–¥–∞–ª–∏—Ç—å ${removedCount} –∑–∞–∫–∞–∑–æ–≤ —Å—Ç–∞—Ä—à–µ ${days} –¥–Ω–µ–π?`)) {
      setOrders(filtered);
    }
  };

  const fetchExchangeRate = async () => {
    try {
      const response = await fetch('https://api.exchangerate-api.com/v4/latest/USD');
      if (!response.ok) throw new Error('rate fetch failed');
      const data = await response.json();
      const rate = data.rates?.UZS || 12650;
      setExchangeRate(rate);
    } catch (error) {
      console.log('Using default rate', error);
      setExchangeRate(12650);
    }
  };

  // filtering
  const term = searchTerm.toLowerCase().trim();
  const filteredProducts = products.filter(p => {
    if (!term) return true;
    return (
      (p.name || '').toLowerCase().includes(term) ||
      (p.id?.toString() || '').includes(term) ||
      (p.category || '').toLowerCase().includes(term)
    );
  });

  const totalPages = Math.max(1, Math.ceil(filteredProducts.length / itemsPerPage));
  const startIndex = (currentPage - 1) * itemsPerPage;
  const paginatedProducts = filteredProducts.slice(startIndex, startIndex + itemsPerPage);

  useEffect(() => setCurrentPage(1), [searchTerm, showAllProducts]);

  // showAllProducts toggle logic
  const visibleProducts = showAllProducts ? filteredProducts : paginatedProducts;

  // cart operations
  const addToCart = (product) => {
    const existingItem = cart.find(item => item.id === product.id);
    if (existingItem) {
      setCart(cart.map(item =>
        item.id === product.id ? { ...item, quantity: item.quantity + 1 } : item
      ));
    } else {
      setCart([...cart, { ...product, quantity: 1 }]);
    }
  };

  const removeFromCart = (productId) => setCart(cart.filter(item => item.id !== productId));

  const updateQuantity = (productId, quantity) => {
    if (quantity <= 0) {
      removeFromCart(productId);
      return;
    }
    setCart(cart.map(item => item.id === productId ? { ...item, quantity } : item));
  };

  const calculateTotal = () => cart.reduce((sum, item) => sum + (item.price * item.quantity), 0);

  const completeOrder = () => {
    if (cart.length === 0) { alert('–ö–æ—Ä–∑–∏–Ω–∞ –ø—É—Å—Ç–∞!'); return; }
    const newOrder = {
      id: Date.now(),
      date: new Date().toISOString(),
      dateFormatted: new Date().toLocaleString('ru-RU'),
      items: JSON.parse(JSON.stringify(cart)),
      total: calculateTotal(),
      customer: { ...customerInfo },
      paymentMethod: '–ù–∞–ª–∏—á–Ω—ã–µ'
    };
    setOrders([newOrder, ...orders]);
    setCart([]);
    setCustomerInfo({ name: '', phone: '' });
    alert(`‚úÖ –ó–∞–∫–∞–∑ #${newOrder.id} –æ—Ñ–æ—Ä–º–ª–µ–Ω! –ò—Ç–æ–≥–æ: $${newOrder.total.toFixed(2)}`);
  };

  const printReceipt = (order) => {
    const printWindow = window.open('', '_blank');
    const receiptHTML = `
      <!DOCTYPE html>
      <html>
      <head><meta charset="utf-8"><title>–ß–µ–∫ #${order.id}</title>
      <style>body{font-family:Arial;max-width:320px;margin:12px;color:#111}.h{text-align:center;border-bottom:1px dashed #222;padding-bottom:8px;margin-bottom:8px}.item{display:flex;justify-content:space-between;margin:6px 0}.total{border-top:1px dashed #222;margin-top:10px;padding-top:8px;font-weight:700}</style>
      </head><body>
      <div class="h"><h3>ELECTROMAX</h3><div>–ß–µ–∫ #${order.id}</div><div>${order.dateFormatted}</div>${order.customer?.name?`<div>–ö–ª–∏–µ–Ω—Ç: ${order.customer.name}</div>`:''}</div>
      ${order.items.map(item => `<div class="item"><div>${item.name} x${item.quantity}</div><div>$${(item.price*item.quantity).toFixed(2)}</div></div>`).join('')}
      <div class="total"><div class="item"><div>–ò–¢–û–ì–û</div><div>$${order.total.toFixed(2)}</div></div><div class="item"><div>–í —Å—É–º</div><div>${(order.total*exchangeRate).toFixed(0)} UZS</div></div></div>
      <div style="text-align:center;margin-top:12px">–°–ø–∞—Å–∏–±–æ –∑–∞ –ø–æ–∫—É–ø–∫—É!</div></body></html>
    `;
    printWindow.document.write(receiptHTML);
    printWindow.document.close();
    setTimeout(() => { printWindow.print(); printWindow.close(); }, 250);
  };

  const deleteOrder = (orderId) => {
    if (window.confirm('–£–¥–∞–ª–∏—Ç—å —ç—Ç–æ—Ç –∑–∞–∫–∞–∑?')) setOrders(orders.filter(o => o.id !== orderId));
  };

  const getFilteredOrders = () => {
    if (!dateFilter.start && !dateFilter.end) return orders;
    return orders.filter(order => {
      const orderDate = new Date(order.date);
      const start = dateFilter.start ? new Date(dateFilter.start) : null;
      const end = dateFilter.end ? new Date(dateFilter.end) : null;
      if (start && orderDate < start) return false;
      if (end && orderDate > end) return false;
      return true;
    });
  };

  const getAnalytics = () => {
    const filteredOrders = getFilteredOrders();
    const totalRevenue = filteredOrders.reduce((sum, o) => sum + o.total, 0);
    const totalOrders = filteredOrders.length;
    const avgOrderValue = totalOrders > 0 ? totalRevenue / totalOrders : 0;
    const today = new Date();
    const todayOrders = filteredOrders.filter(o => new Date(o.date).toDateString() === today.toDateString());
    const todayRevenue = todayOrders.reduce((sum, o) => sum + o.total, 0);
    const productSales = {};
    filteredOrders.forEach(order => {
      order.items.forEach(item => {
        if (!productSales[item.name]) productSales[item.name] = { quantity: 0, revenue: 0 };
        productSales[item.name].quantity += item.quantity;
        productSales[item.name].revenue += item.price * item.quantity;
      });
    });
    const topProducts = Object.entries(productSales).sort((a, b) => b[1].revenue - a[1].revenue).slice(0, 10);
    return { totalRevenue, totalOrders, avgOrderValue, todayRevenue, todayOrders: todayOrders.length, topProducts };
  };

  const exportOrders = () => {
    const dataStr = JSON.stringify(orders, null, 2);
    const blob = new Blob([dataStr], { type: 'application/json' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `orders_${new Date().toISOString().split('T')[0]}.json`;
    document.body.appendChild(a);
    a.click();
    a.remove();
    URL.revokeObjectURL(url);
  };

  const downloadProductsJSON = (updatedProducts) => {
    const jsonData = JSON.stringify({ products: updatedProducts }, null, 2);
    const blob = new Blob([jsonData], { type: 'application/json' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = 'products.json';
    document.body.appendChild(a);
    a.click();
    a.remove();
    URL.revokeObjectURL(url);
  };

  const saveProduct = (productId) => {
    const updated = products.map(p => p.id === productId ? { ...p, name: editForm.name, price: parseFloat(editForm.price), unit: editForm.unit } : p);
    setProducts(updated);
    downloadProductsJSON(updated);
    alert('üíæ –ò–∑–º–µ–Ω–µ–Ω–∏—è —Å–æ—Ö—Ä–∞–Ω–µ–Ω—ã! –°–∫–∞—á–∞–Ω products.json');
    setEditingProduct(null);
  };

  // inline card edit handlers
  const startCardEdit = (product) => {
    setCardEditingId(product.id);
    setCardEditForm({ name: product.name || '', price: String(product.price || ''), unit: product.unit || '' });
  };
  const cancelCardEdit = () => {
    setCardEditingId(null);
    setCardEditForm({ name: '', price: '', unit: '' });
  };
  const saveCardEdit = (productId) => {
    const updated = products.map(p => p.id === productId ? { ...p, name: cardEditForm.name, price: parseFloat(cardEditForm.price) || 0, unit: cardEditForm.unit } : p);
    setProducts(updated);
    downloadProductsJSON(updated);
    alert('üíæ –ò–∑–º–µ–Ω–µ–Ω–∏—è —Å–æ—Ö—Ä–∞–Ω–µ–Ω—ã! –°–∫–∞—á–∞–Ω products.json');
    cancelCardEdit();
  };

  const analytics = getAnalytics();

  if (loading) {
    return (
      <div className="min-h-screen bg-gradient-to-br from-slate-950 via-slate-900 to-zinc-900 flex items-center justify-center">
        <div className="text-center">
          <div className="animate-spin rounded-full h-16 w-16 border-t-4 border-b-4 border-emerald-500 mx-auto mb-4"></div>
          <p className="text-lg font-medium text-slate-100">–ó–∞–≥—Ä—É–∑–∫–∞ —Å–∏—Å—Ç–µ–º—ã...</p>
        </div>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-gradient-to-br from-slate-950 via-slate-900 to-zinc-900 text-slate-100">
      <header className="bg-gradient-to-r from-slate-900 to-slate-800 shadow-2xl border-b border-slate-700">
        <div className="max-w-full px-6 py-4">
          <div className="flex items-center justify-between gap-4">
            <div className="flex items-center gap-4">
              <div className="bg-gradient-to-br from-emerald-500 to-emerald-600 p-3 rounded-xl shadow-lg">
                <Package className="h-8 w-8 text-white" />
              </div>
              <div>
                <h1 className="text-white font-bold text-2xl tracking-wide">ELECTROMAX PRO</h1>
                <p className="text-slate-400 text-sm">–ü—Ä–æ—Ñ–µ—Å—Å–∏–æ–Ω–∞–ª—å–Ω–∞—è –∫–∞—Å—Å–æ–≤–∞—è —Å–∏—Å—Ç–µ–º–∞</p>
              </div>
            </div>

            <div className="flex items-center gap-4">
              <div className="bg-slate-800 px-4 py-2 rounded-lg border border-slate-700 text-center">
                <p className="text-xs text-slate-400">–¢–æ–≤–∞—Ä–æ–≤</p>
                <p className="text-xl font-bold text-emerald-400">{products.length}</p>
              </div>

              <div className="bg-slate-800 px-4 py-2 rounded-lg border border-slate-700 text-center">
                <p className="text-xs text-slate-400">–ó–∞–∫–∞–∑–æ–≤</p>
                <p className="text-xl font-bold text-blue-400">{orders.length}</p>
              </div>

              <div className="bg-slate-800 px-4 py-2 rounded-lg border border-slate-700 text-center">
                <p className="text-xs text-slate-400">–ö—É—Ä—Å USD</p>
                <p className="text-sm font-bold text-yellow-400">{Number(exchangeRate).toLocaleString()}</p>
              </div>

              <div className="hidden md:flex items-center bg-slate-800 rounded-lg px-3 py-2">
                <Search className="h-4 w-4 text-slate-400" />
                <input
                  type="text"
                  placeholder="–ü–æ–∏—Å–∫..."
                  value={searchTerm}
                  onChange={(e) => setSearchTerm(e.target.value)}
                  className="bg-transparent outline-none text-sm text-slate-100 ml-3"
                />
                <button onClick={() => setShowAllProducts(v => !v)} className="ml-3 px-2 py-1 bg-slate-700 rounded text-xs">{showAllProducts ? '–ü–æ—Å—Ç—Ä–∞–Ω–∏—á–Ω–æ' : '–ü–æ–∫–∞–∑–∞—Ç—å –≤—Å–µ'}</button>
              </div>

              <button onClick={() => setShowCalculator(!showCalculator)} className="bg-gradient-to-r from-purple-600 to-purple-700 px-3 py-2 rounded-lg">
                <Calculator className="h-5 w-5 text-white" />
              </button>
            </div>
          </div>
        </div>
      </header>

      {showCalculator && (
        <div className="fixed inset-0 bg-black bg-opacity-50 flex items-center justify-center z-50" onClick={() => setShowCalculator(false)}>
          <div className="bg-slate-800 rounded-2xl p-6 shadow-2xl border border-slate-700 w-96" onClick={e => e.stopPropagation()}>
            <div className="flex justify-between items-center mb-4">
              <h3 className="text-xl font-bold text-white flex items-center gap-2"><Calculator className="h-5 w-5 text-purple-400" /> –ö–æ–Ω–≤–µ—Ä—Ç–µ—Ä –≤–∞–ª—é—Ç</h3>
              <button onClick={() => setShowCalculator(false)} className="text-slate-400"><X className="h-5 w-5" /></button>
            </div>

            <input type="number" value={calculatorAmount} onChange={(e) => setCalculatorAmount(e.target.value)} placeholder="–°—É–º–º–∞" className="w-full px-3 py-2 bg-slate-900 rounded mb-3" />

            <div className="flex gap-2 mb-3">
              <button onClick={() => setSelectedCurrency('USD')} className={`flex-1 py-2 rounded ${selectedCurrency === 'USD' ? 'bg-purple-600 text-white' : 'bg-slate-700 text-slate-300'}`}>USD ‚Üí UZS</button>
              <button onClick={() => setSelectedCurrency('UZS')} className={`flex-1 py-2 rounded ${selectedCurrency === 'UZS' ? 'bg-purple-600 text-white' : 'bg-slate-700 text-slate-300'}`}>UZS ‚Üí USD</button>
            </div>

            {calculatorAmount && (
              <div className="bg-gradient-to-r from-purple-900 to-purple-800 p-4 rounded text-center">
                <div className="text-sm text-purple-200 mb-1">–†–µ–∑—É–ª—å—Ç–∞—Ç</div>
                <div className="text-2xl font-bold text-white">
                  {selectedCurrency === 'USD'
                    ? `${(parseFloat(calculatorAmount) * exchangeRate).toLocaleString()} UZS`
                    : `$${(parseFloat(calculatorAmount) / exchangeRate).toFixed(2)}`
                  }
                </div>
              </div>
            )}
          </div>
        </div>
      )}

      <nav className="bg-slate-800 border-b border-slate-700">
        <div className="flex overflow-x-auto">
          {[
            { id: 'order', label: '–ù–æ–≤—ã–π –∑–∞–∫–∞–∑', icon: ShoppingCart },
            { id: 'products', label: '–¢–æ–≤–∞—Ä—ã', icon: Package },
            { id: 'orders', label: '–ò—Å—Ç–æ—Ä–∏—è', icon: FileText },
            { id: 'analytics', label: '–ê–Ω–∞–ª–∏—Ç–∏–∫–∞', icon: BarChart3 }
          ].map(view => (
            <button
              key={view.id}
              onClick={() => setCurrentView(view.id)}
              className={`flex-1 min-w-max py-4 px-6 font-semibold text-sm transition-all flex items-center justify-center gap-2 ${currentView === view.id ? 'bg-emerald-600 text-white' : 'bg-slate-800 text-slate-300 hover:bg-slate-700'}`}
            >
              <view.icon className="h-5 w-5" /> {view.label}
            </button>
          ))}
        </div>
      </nav>

      <div className="max-w-7xl mx-auto px-4 py-6">
        {currentView === 'order' && (
          <div className="grid grid-cols-1 lg:grid-cols-3 gap-6">
            <div className="lg:col-span-2">
              <div className="bg-slate-800 rounded-2xl border border-slate-700 p-6">
                <h2 className="text-2xl font-bold text-white mb-4 flex items-center gap-2"><Package className="h-6 w-6 text-emerald-400" /> –ö–∞—Ç–∞–ª–æ–≥ —Ç–æ–≤–∞—Ä–æ–≤</h2>

                <div className="relative mb-4 md:hidden">
                  <Search className="absolute left-3 top-3.5 h-5 w-5 text-slate-400" />
                  <input type="text" placeholder="–ü–æ–∏—Å–∫..." value={searchTerm} onChange={e => setSearchTerm(e.target.value)} className="w-full px-4 py-3 pl-10 bg-slate-900 border border-slate-600 rounded-lg text-white" />
                </div>

                <div className="space-y-2 max-h-96 overflow-y-auto">
                  {visibleProducts.map(product => (
                    <div key={product.id} className="flex items-center justify-between p-4 bg-slate-900 rounded-lg border border-slate-700">
                      <div className="min-w-0">
                        {cardEditingId === product.id ? (
                          <>
                            <input value={cardEditForm.name} onChange={e => setCardEditForm(f => ({ ...f, name: e.target.value }))} className="w-full bg-slate-800 px-2 py-1 rounded mb-2" />
                            <div className="flex gap-2">
                              <input value={cardEditForm.price} onChange={e => setCardEditForm(f => ({ ...f, price: e.target.value }))} className="w-28 bg-slate-800 px-2 py-1 rounded" />
                              <input value={cardEditForm.unit} onChange={e => setCardEditForm(f => ({ ...f, unit: e.target.value }))} className="w-24 bg-slate-800 px-2 py-1 rounded" />
                            </div>
                          </>
                        ) : (
                          <>
                            <p className="text-white font-semibold truncate">{product.name}</p>
                            <p className="text-xs text-slate-400">ID: {product.id} ‚Ä¢ {product.category}</p>
                          </>
                        )}
                      </div>

                      <div className="flex items-center gap-3">
                        <div className="text-right">
                          <p className="text-xl font-bold text-white">${product.price}</p>
                          <p className="text-xs text-slate-400">{(product.price * exchangeRate).toFixed(0)} UZS</p>
                        </div>

                        {cardEditingId === product.id ? (
                          <>
                            <button onClick={() => saveCardEdit(product.id)} className="px-3 py-2 bg-cyan-500 rounded text-black">–°–æ—Ö—Ä–∞–Ω–∏—Ç—å</button>
                            <button onClick={cancelCardEdit} className="px-3 py-2 bg-slate-700 rounded">–û—Ç–º–µ–Ω–∞</button>
                          </>
                        ) : (
                          <>
                            <button onClick={() => startCardEdit(product)} className="px-3 py-2 bg-blue-600 text-white rounded-lg"><Edit2 className="h-4 w-4" /></button>
                            <button onClick={() => addToCart(product)} className="p-3 bg-emerald-600 hover:bg-emerald-700 text-white rounded-lg"><Plus className="w-5 h-5" /></button>
                          </>
                        )}
                      </div>
                    </div>
                  ))}
                </div>

                {!showAllProducts && totalPages > 1 && (
                  <div className="flex justify-center gap-2 mt-4">
                    <button onClick={() => setCurrentPage(p => Math.max(1, p - 1))} disabled={currentPage === 1} className="px-4 py-2 bg-slate-700 text-white rounded-lg disabled:opacity-50"><ArrowLeft className="h-4 w-4" /></button>
                    <span className="px-4 py-2 bg-emerald-600 text-white rounded-lg">{currentPage} / {totalPages}</span>
                    <button onClick={() => setCurrentPage(p => Math.min(totalPages, p + 1))} disabled={currentPage === totalPages} className="px-4 py-2 bg-slate-700 text-white rounded-lg disabled:opacity-50"><ArrowRight className="h-4 w-4" /></button>
                  </div>
                )}
              </div>
            </div>

            <div>
              <div className="bg-slate-800 rounded-2xl border border-slate-700 p-6 sticky top-6">
                <h2 className="text-2xl font-bold text-white mb-4 flex items-center gap-2"><ShoppingCart className="h-6 w-6 text-emerald-400" /> –ö–æ—Ä–∑–∏–Ω–∞ {cart.length > 0 && <span className="bg-emerald-600 text-sm px-2 py-1 rounded-full">{cart.length}</span>}</h2>

                <div className="mb-4">
                  <input type="text" placeholder="–ò–º—è –∫–ª–∏–µ–Ω—Ç–∞" value={customerInfo.name} onChange={e => setCustomerInfo({ ...customerInfo, name: e.target.value })} className="w-full px-4 py-2 bg-slate-900 border border-slate-600 rounded-lg text-white mb-2" />
                  <input type="tel" placeholder="–¢–µ–ª–µ—Ñ–æ–Ω" value={customerInfo.phone} onChange={e => setCustomerInfo({ ...customerInfo, phone: e.target.value })} className="w-full px-4 py-2 bg-slate-900 border border-slate-600 rounded-lg text-white" />
                </div>

                  </>

            </div>
          </div>
        )}

        {currentView === 'products' && (

            <h2 className="text-2xl font-bold text-white mb-4">–£–ø—Ä–∞–≤–ª–µ–Ω–∏–µ —Ç–æ–≤–∞—Ä–∞–º–∏</h2>

              <table className="w-full">

                      </td>
                    </tr>
                  ))}
                </tbody>
              </table>
            </div>
          </div>
        )}

        {currentView === 'orders' && (
          <div className="bg-slate-800 rounded-2xl border border-slate-700 p-6">
            <div className="flex items-center justify-between mb-4">
              <h2 className="text-2xl font-bold text-white">–ò—Å—Ç–æ—Ä–∏—è –∑–∞–∫–∞–∑–æ–≤</h2>

                  </div>
                ))}
              </div>
            )}
          </div>
        )}

        {currentView === 'analytics' && (
          <div className="space-y-6">
            <div className="bg-slate-800 rounded-2xl border border-slate-700 p-6">
              <h2 className="text-2xl font-bold text-white mb-6">–ê–Ω–∞–ª–∏—Ç–∏–∫–∞ –ø—Ä–æ–¥–∞–∂</h2>

                  <DollarSign className="h-8 w-8 text-emerald-400 mb-2" />

                  <div className="space-y-3">
                    {analytics.topProducts.map(([name, data], idx) => (

                          <p className="text-lg font-bold text-white">${data.revenue.toFixed(2)}</p>

                      </div>
                    ))}
                  </div>
                )}
              </div>
            </div>
          </div>
        )}
      </div>
    </div>
  );
};
